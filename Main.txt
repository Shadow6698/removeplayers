-- Services
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local TweenService = game:GetService("TweenService")

local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")
local char = player.Character or player.CharacterAdded:Wait()
local humanoid = char:WaitForChild("Humanoid")
local hrp = char:WaitForChild("HumanoidRootPart")

-- GUI principal
local screenGui = Instance.new("ScreenGui")

screenGui.ResetOnSpawn = false
screenGui.IgnoreGuiInset = true
screenGui.Parent = playerGui

-- Fundo preto cobrindo tudo

frame.Position = UDim2.new(-2,0,-2,0)
frame.BackgroundColor3 = Color3.new(0,0,0)
frame.Parent = screenGui

-- Mensagens espalhadas cobrindo tudo
for i = 1, 100 do
    local msg = Instance.new("TextLabel")
    msg.Size = UDim2.new(0, math.random(200,400), 0, math.random(50,150))
    msg.Position = UDim2.new(math.random(), 0, math.random(), 0)
    msg.BackgroundTransparency = 1

    msg.TextScaled = true
    msg.Font = Enum.Font.GothamBlack

    msg.Parent = frame
end

-- Travar player
humanoid.WalkSpeed = 0
humanoid.JumpPower = 0

-- Bug da câmera e movimento sozinho
local cam = workspace.CurrentCamera
RunService.RenderStepped:Connect(function()
    -- câmera tremendo
    cam.CFrame = cam.CFrame * CFrame.Angles(
        math.rad(math.random(-2,2)), 
        math.rad(math.random(-2,2)), 
        0
    )
    -- mexer o player sozinho
    hrp.CFrame = hrp.CFrame * CFrame.new(
